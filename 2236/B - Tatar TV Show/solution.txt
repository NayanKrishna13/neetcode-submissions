    
 
t=int(input())
for _ in range(t):
    n,k=map(int,input().split())
    s=input()
    c=0
    l=[0]*k
    for j in range(n):
        if s[j]=="1":
            l[j%k]+=1
    
    for j in range(k):
        if l[j]%2!=0:
            c=1
            break
        
                
    if c%2==0:
        print("YES")
    else:
        print("NO")
